JavaScript Cheatsheet

Variables

const name = "Juan";   // cannot reassign (default!)
let age = 25;          // can reassign
// var — avoid!

Data Types

let str = "text";
let num = 42;
let bool = true;
let arr = [1, 2, 3];
let obj = { key: "value" };

Functions

// Declaration
function add(a, b) { return a + b; }

// Arrow function
const add = (a, b) => a + b;

// Default params
const greet = (name = "Guest") => `Hello ${name}`;

DOM

const el = document.querySelector('.class');
const all = document.querySelectorAll('p');
el.textContent = 'new text';
el.innerHTML = '<em>HTML</em>';
el.style.color = 'red';
el.classList.add('active');
el.classList.toggle('active');

Events

el.addEventListener('click', () => {
  console.log('clicked!');
});

Common: click, submit, mouseover, keydown, load

Conditionals

if (condition) {
  // ...
} else if (other) {
  // ...
} else {
  // ...
}

condition ? 'true case' : 'false case';  // ternary

Arrays

arr.length           // length
arr.push(item)       // add to end
arr.pop()            // remove from end
arr.map(fn)          // transform each
arr.filter(fn)       // keep matches
arr.find(fn)         // first match
arr.forEach(fn)      // loop

Async / Fetch

async function getData() {
  const res = await fetch(url);
  const data = await res.json();
  return data;
}

Comparison

===  // strict equal (USE THIS)
!==  // strict not equal
==   // loose equal (AVOID)
!=   // loose not equal (AVOID)
